# Question 59. Spiral Matrix II

Given a positive integer `n`, generate an `n x n` matrix filled with elements from `1` to `n * n` in spiral order.

```python
class Solution(object):
    
    def generateMatrix(self, n):
        """
        :type n: int
        :rtype: List[List[int]]
        """
        # Initialise an n x n matrix filled by None.
        result = [[None for i in range(n)] for j in range(n)]
        # Four directions: Right, Down, Left, Up, in order.
        direction = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        # Use curr_loc to track the entry in the matrix to fill.
        curr_loc = (0, 0)
        # Use fill to track the number to be filled.
        fill = 1
        while fill <= (n * n):
            # Fill the entry of the matrix at curr_loc
            result[curr_loc[0]][curr_loc[1]] = fill
            # Update the number to fill.
            fill += 1
            # Determine the next entry to fill the value
            next_loc = (curr_loc[0] + direction[0][0], curr_loc[1] + direction[0][1])
            if not (-1 < next_loc[0] < n) or \ # Out of boundary
                not (-1 < next_loc[1] < n) or \ # Out of boundary
                result[next_loc[0]][next_loc[1]]: # Already filled
                # Update the diretion to move and update curr_loc
                direction.insert(3, direction.pop(0))
                curr_loc = (curr_loc[0] + direction[0][0], curr_loc[1] + direction[0][1])
            else:
                # The next_loc is correct. Update curr_loc with next_loc.
                curr_loc = next_loc     
        return result
```
